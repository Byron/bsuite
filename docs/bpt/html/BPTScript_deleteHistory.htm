string $hilited[] = `ls -hl`;
string $selected[] = `ls -sl`;

string $dep[] = `ls -sl -dep`;
string $dagRemovals[] = `ls -sl -dag`;

$dep = stringArrayRemove($dagRemovals, $dep);
$selected = stringArrayRemove($dep, $selected); 
 


if(size($hilited) > 0)
	select $hilited;
else
	select $selected;

delete -ch;

select $selected;
hilite $hilited;
